# ការបញ្ជាក់លេខកូដ JS

## រចនាប័ទ្មសរសេរកូដ

- សាមញ្ញ ឆើតឆាយ និងមានប្រសិទ្ធភាព ដោយប្រើវាក្យសម្ព័ន្ធ js ទំនើបបំផុត។
- បំបែកមុខងារវែងពេក ហើយការរចនាចំណុចប្រទាក់គួរតែជាគូភ្ជាប់ទាប និងស្អិតរមួតខ្ពស់។
- ប្រើ arrays + loops/forEach/map កាន់តែច្រើន ហើយសរសេរកូដដដែលៗតិច
- កុំសរសេរមតិយោបល់មុខងារជាកូដសាមញ្ញ ដើម្បីជៀសវាងការបញ្ចេញមតិច្រើនពេក។ តក្កវិជ្ជាស្មុគស្មាញ និងតម្រូវការពិសេសត្រូវបានពន្យល់ជាភាសាចិនចម្រាញ់។
- កូដដដែលៗត្រូវបានអរូបីចូលទៅក្នុងមុខងារ ដើម្បីកាត់បន្ថយភាពមិនដូចគ្នា ឧទាហរណ៍ `cosnt A=new Table({ style: { border: NO_BORDER } }), B=new Table({ style: { border: NO_BORDER } })` អាច​ត្រូវ​បាន​សរសេរ​ជា `const newTable=()=>new Table({ style: { border: NO_BORDER } }), A=newTable(), B=newTable()`
- ការ​ប្រកាស​អថេរ៖ ផ្សំ​ការ​ប្រកាស `const` ច្រើន​ជាប់​គ្នា​ទៅ​ក្នុង​សេចក្តី​ថ្លែងការណ៍​មួយ។ ជំនួសឱ្យការសរសេរវាជាបីបន្ទាត់ សូមសរសេរ `const a=1, b=2, c=3;`
- ដំណើរការអសមកាល៖ ប្រើ `await` ហាមប្រើ`.then`ការហៅជាខ្សែសង្វាក់
- កុំបង្កើតកូដដោះស្រាយករណីលើកលែងដោយស្វ័យប្រវត្តិ កុំសរសេរដោយស្វ័យប្រវត្តិ `try...catch` (`try ... catch`ត្រូវបានរក្សាដោយដៃ។ `try catch`ដែលមានស្រាប់ មិនគួរត្រូវបានលុបទេ)
- មុខងារសុទ្ធដំបូង៖ សរសេរតែមុខងារសុទ្ធប៉ុណ្ណោះ មិនដែលថ្នាក់ទេ។
- ប្រើមុខងារព្រួញ `const funcName = () => {}` ហើយកុំប្រើពាក្យគន្លឹះ `function` (លើកលែងតែម៉ាស៊ីនភ្លើង); ប្រសិនបើអ្នកអាចប្រើ .bind ដើម្បីចងប៉ារ៉ាម៉ែត្រ ជៀសវាងការកំណត់មុខងារ
- ការប្រើប្រាស់កូដឡើងវិញ៖ យកចិត្តទុកដាក់ចំពោះការប្រើប្រាស់ឡើងវិញ ទាញយកមុខងារតូចៗបន្ថែមទៀត និងយ៉ាងម៉ឺងម៉ាត់ជៀសវាងនូវរចនាសម្ព័ន្ធកូដស្រដៀងគ្នា ឬចម្លងដែលបានបិទភ្ជាប់មួយចំនួនធំ។
- ការចូលប្រើវត្ថុ៖ កំណត់អាទិភាពដោយប្រើការចាត់តាំងការបំផ្លាញដើម្បីទាញយកគុណលក្ខណៈដែលត្រូវការ ជៀសវាងការប្រើចំណុចៗម្តងហើយម្តងទៀតនៅខាងក្នុងដើម្បីចូលប្រើគុណលក្ខណៈជ្រៅ និងជាប់គ្នា ហើយបញ្ចូលការវិនិច្ឆ័យខ្សែសង្វាក់ជម្រើសម្តងហើយម្តងទៀត។
- កុំប្រើវត្ថុសម្រាប់ប៉ារ៉ាម៉ែត្រមុខងារ កុំសរសេរ `{a,b,c}` សរសេរ `a, b, c`; ប្រសិនបើមានប៉ារ៉ាម៉ែត្រស្រេចចិត្តច្រើន ប្រើ [[ចំនួនធាតុកំណត់រចនាសម្ព័ន្ធ ធាតុកំណត់រចនាសម្ព័ន្ធ] [ចំនួនធាតុកំណត់រចនាសម្ព័ន្ធ] ធាតុកំណត់រចនាសម្ព័ន្ធ], ..] ក្នុងគំរូនេះ ធាតុកំណត់រចនាសម្ព័ន្ធត្រូវបានកំណត់ដោយលេខថេរ។
- សម្រាប់​តម្លៃ​ត្រឡប់​ច្រើន ប្រើ​អារេ `[a,b,c]` ជំនួសឱ្យ `{a,b,c}`។ ប្រសិនបើមានតម្លៃត្រឡប់ច្រើនជាង 3 ប្រើថេរជាលេខដើម្បីកំណត់អត្ថន័យទីតាំង។
- មិនចាំបាច់មានខ្សែអក្សរដើម្បីតំណាងឱ្យរដ្ឋទេ ប្រើលេខថេរដើម្បីកំណត់ស្ថានភាព
- គ្មានគំរូខ្សែអក្សរ (``) ត្រូវបានប្រើប្រាស់ ការភ្ជាប់ខ្សែអក្សរ (+) ត្រូវបានប្រើ លើកលែងតែការនាំចូល (ងាយស្រួលសម្រាប់ការវិភាគឋិតិវន្ត Vite)
- សម្រាប់រង្វិលជុំ ប្រសិនបើអ្នកត្រូវការលេខស៊េរី សូមប្រើ `++i`ជំនួសឱ្យ `i++`
- ជៀសវាង`?.`ដែលមិនចាំបាច់ និងហាមឃាត់ការសរសេរកម្មវិធីការពារ
- ឈ្មោះអថេរមិនចាំបាច់ជាពហុវចនៈទេ ហើយបញ្ចប់ដោយ `_li` ដើម្បីបង្ហាញបញ្ជី។

## អនុសញ្ញាឈ្មោះ

- ការ​ដាក់​ឈ្មោះ​បន្ត​ទៅ​តាម​ភាព​តិចតួច​បំផុត។ ប្រើឈ្មោះខ្លីៗ ប៉ុន្តែមានអត្ថន័យ ឧទាហរណ៍៖ ប្រើ `rm`ជំនួសឱ្យ `remove`, `delete`, `del`។ ទោះយ៉ាងណាក៏ដោយ អ្នកក៏គួរតែជៀសវាងការហួសហេតុផងដែរ ឧទាហរណ៍៖ កុំប្រើអក្សរតែមួយ `m`ជំនួសឱ្យ `map`
- ព្យាយាមប្រើតែកិរិយាស័ព្ទក្នុងការដាក់ឈ្មោះមុខងារ។ ប្រសិនបើអ្នកអាចប្រើពាក្យមួយដើម្បីបង្ហាញវា សូមកុំប្រើពាក្យពីរ។ នាមត្រូវបានឆ្លុះបញ្ចាំងនៅក្នុងឈ្មោះឯកសារ។ ប្រសិនបើអ្នកត្រូវការបន្ថែមកិរិយាស័ព្ទទៅឈ្មោះឯកសារ សូមដាក់នាមមុន និងកិរិយាស័ព្ទចុងក្រោយ។ ឧទាហរណ៍៖ `profileSet.js` ជំនួសឱ្យ `setProfile.js`
- ឈ្មោះអថេរ៖ ប្រើរចនាប័ទ្មគូសបន្ទាត់ពីក្រោម (snake_case) ឧទាហរណ៍ `user_auth_token`; ប្រសិនបើអថេរគឺជាអនុគមន៍ ប្រើរចនាប័ទ្មដាក់ឈ្មោះអក្សរតូច
- ឈ្មោះមុខងារ៖ ប្រើអក្សរតូច camelCase (camelCase) ឧទាហរណ៍ `userData`
- ប៉ារ៉ាម៉ែត្រមុខងារ៖ ប្រសិនបើវាជាមុខងារហៅត្រឡប់មកវិញ សូមដាក់ឈ្មោះវាក្នុងអក្សរតូចអូដ្ឋ ដូចជា `onChange`
- និយមន័យថេរកម្រិតម៉ូឌុល ប្រើរចនាប័ទ្មបន្ទាត់ក្រោមអក្សរធំ `UPPER_SNAKE_CASE`
- កុំសរសេរ `get` បុព្វបទគ្មានន័យដូចជា៖ សរសេរ `cookieByHeader`ជំនួសឱ្យ `getCookie`
- អថេរកម្រិតម៉ូឌុល/សកល៖ ប្រើរចនាប័ទ្មបន្ទាត់ក្រោមអក្សរធំ (UPPER_SNAKE_CASE) ឧ. `CODE_TO_ID`, `ID_TO_LANG`

## យន្តការម៉ូឌុល

- នាំចូល៖ មុខងារនាំចូលត្រឹមត្រូវតាមតម្រូវការ ហាមឃាត់ការនាំចូលដោយផ្ទាល់នៃម៉ូឌុលទាំងមូល (ជៀសវាង `import * as x` ឬនាំចូលវត្ថុធំ)
- ការនាំចេញ៖ បិទការនាំចេញវត្ថុ។ នាំចេញអនុគមន៍ និងអថេរជាឯកតា ដូចជា `export const X=1, abc=()=>{};`។ ព្យាយាមបញ្ចូលពួកវាជាមួយ const + សញ្ញាក្បៀស ដើម្បីប្រកាសមាតិកាដែលបាននាំចេញ។ ប្រសិនបើឯកសារមានមុខងារតែមួយ សូមប្រើ `export default`
- លុះត្រាតែអ្នកចាំបាច់ត្រូវហៅមុខងារលំនាំដើមនាំចេញនៅខាងក្នុង ជៀសវាងការប្រកាសតម្លៃថេរជាមុនសិន ហើយបន្ទាប់មកនាំចេញពួកវានៅចុងបញ្ចប់នៃឯកសារ។
- ដំណោះស្រាយផ្លូវ៖ នៅពេលទទួលបានផ្លូវថតបច្ចុប្បន្ន `import.meta.dirname` ត្រូវតែប្រើ
