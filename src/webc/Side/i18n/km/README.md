# бһ”бһ“бҹ’бһ‘бҹҮбһҖбҹ’бһҸбһ¶бһҡбһҠбҹӮбһӣбһҡбһ»бһүбһ…бҹҒбһүбһ–бһёбһ…бҹҶбһ бҹҖбһ„

- бһ”бһ„бҹ’бһҖбһҫбһҸбһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’бһҡбһҳбһјбһҡбһҒбһ¶бһ„бһҖбҹ’бһ“бһ»бһ„бһҠбҹ„бһҷбһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбһ·
- бһ…бһ»бһ…вҖӢбһҸбҹҶбһ”бһ“бҹӢвҖӢбһҒбһ¶бһ„вҖӢбһҖбҹ’бһҡбҹ…вҖӢбһҠбһҫбһҳбҹ’бһ”бһёвҖӢбһ”бһ„бҹ’бһҡбһҪбһҳвҖӢбһҠбҹ„бһҷвҖӢбһҹбҹ’бһңбҹҗбһҷвҖӢбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбһ·
- бһҖбһ¶бһҷбһңбһ·бһҖбһ¶бһҡбһўбһјбһҹбһ‘бҹ…бһҶбҹ’бһңбҹҒбһ„ бһ“бһ·бһ„бһҹбҹ’бһҸбһ¶бҹҶ бһҠбһҫбһҳбҹ’бһ”бһёбһ”бһ·бһ‘
- бһ‘бһёбһҸбһ¶бҹҶбһ„бһҡбһўбһ·бһӣбһ…бҹҶбһ бҹҖбһ„бһҖбҹ’бһ“бһ»бһ„бһ‘бһ·бһҹбһҠбҹ…бһҶбҹ’бһңбҹҒбһ„бһ“бһ·бһ„бһҹбҹ’бһҸбһ¶бҹҶ
- бһ•бҹ’бһҸбһӣбҹӢбһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚбһ”бһҫбһҖ бһ“бһ·бһ„бһ”бһ·бһ‘

## бһ”бҹ’бһҡбһҫбһҖбһ¶бһҡбһ”бһ„бҹ’бһ бһ¶бһү

```html
<button class="open-btn">бһ”бһҫбһҖ</button>

<c-side class="right">
  <div>бһҳбһ¶бһҸбһ·бһҖбһ¶бһҡбһ”бһҹбҹӢбһҗбһҸ</div>
</c-side>

<script type="module">
import { open } from "webc.site/Side.js";

const btn = document.querySelector(".open-btn"),
  side = document.querySelector("c-side");

btn.onclick = () => open(side);
</script>
```

## бһ…бҹҶбһҺбһ»бһ…бһ”бҹ’бһҡбһ‘бһ¶бһҖбҹӢ

### бһңбһ·бһ’бһёбһҹбһ¶бһҹбҹ’бһҡбҹ’бһҸ

- `open(el)`бҹ– бһҳбһ»бһҒбһ„бһ¶бһҡбһ“бһ¶бҹҶбһ…бҹҒбһү бһ”бһҫбһҖбһҗбһҸ
- `close(el)`бҹ– бһҳбһ»бһҒбһ„бһ¶бһҡбһ“бһ¶бҹҶбһ…бҹҒбһү бһ”бһ·бһ‘бһҗбһҸ
- `el.open()`бҹ– бһңбһ·бһ’бһёбһҹбһ¶бһҹбҹ’бһҸбҹ’бһҡбһ§бһ‘бһ¶бһ бһҡбһҺбҹҚ бһ”бһҫбһҖбһҗбһҸ
- `el.close()`бҹ– бһңбһ·бһ’бһёбһҹбһ¶бһҹбҹ’бһҸбҹ’бһҡбһ§бһ‘бһ¶бһ бһҡбһҺбҹҚ бһ”бһ·бһ‘бһҗбһҸ

### бһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚ

- `open`бҹ– бһҶбҹҒбҹҮбһ“бҹ…бһ–бҹҒбһӣбһ”бһҫбһҖ
- `close`бҹ– бһҶбҹҒбҹҮбһ“бҹ…бһ–бҹҒбһӣбһ”бһ·бһ‘

## бһҗбҹ’бһ“бһ¶бһҖбҹӢбһҡбһ…бһ“бһ¶бһ”бҹҗбһ‘бҹ’бһҳ

- `.right`бҹ– бһҖбһ¶бһҡбһҖбҹҶбһҺбһҸбҹӢбһ‘бһёбһҸбһ¶бҹҶбһ„бһҡбһ»бһүбһ…бҹҒбһүбһҒбһ¶бһ„бһҹбҹ’бһҸбһ¶бҹҶ
- `.top`бҹ– бһҡбһ»бһүбһ…бҹҒбһүбһ‘бһёбһҸбһ¶бҹҶбһ„
- `.bottom`бҹ– бһҡбһ»бһүбһ…бҹҒбһүбһ–бһёбһ‘бһёбһҸбһ¶бҹҶбһ„
- (бһ”бһҫбһҳбһ·бһ“бһҖбҹҶбһҺбһҸбҹӢбһ‘бҹҒ бһӣбҹҶбһ“бһ¶бҹҶбһҠбһҫбһҳбһӮбһәбһҸбҹ’бһҡбһјбһңбһҡбҹҶбһҖбһ·бһӣбһ‘бҹ…бһҒбһ¶бһ„бһҶбҹ’бһңбҹҒбһ„)
- `.open`бҹ– бһҹбҹ’бһҗбһ¶бһ“бһ—бһ¶бһ–бһ”бһҫбһҖ
- `.H`бҹ– бһҹбҹ’бһҗбһ¶бһ“бһ—бһ¶бһ–бһӣбһ¶бһҖбҹӢ
- `.T`бҹ– бһҹбҹ’бһҗбһ¶бһ“бһ—бһ¶бһ–бһ•бҹ’бһӣбһ¶бһҹбҹӢбһ”бҹ’бһҸбһјбһҡбһ…бһӣбһ“бһ¶
