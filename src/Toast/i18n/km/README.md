# бһҮбһҳбҹ’бһҡбһ»бһүбһ”бһ„бҹ’бһўбһҪбһ…бһӣбҹҒбһ…бһЎбһҫбһ„бһ“бҹ…бһӮбҹӮбһҳбһ‘бҹҶбһ–бҹҗбһҡ

- **бһҖбһ¶бһҡбһ бҹ…бһҳбһ»бһҒбһ„бһ¶бһҡ**бҹ– бһ•бҹ’бһҸбһӣбҹӢбһ“бһјбһңбһ…бҹҶбһҺбһ»бһ…бһ”бҹ’бһҡбһ‘бһ¶бһҖбҹӢбһ”бҹ’бһҡбһўбһ”бҹӢбһ”бһүбҹ’бһ…бһјбһӣбһ’бһҳбҹ’бһҳбһҸбһ¶ бһ“бһ·бһ„бһҖбҹҶбһ бһ»бһҹ
- ** бһҮбһ„бҹӢбһҠбҹ„бһҷбһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбһ· **бҹ– бһӮбһ“бҹ’бһӣбһ№бҹҮбһҮбһ¶бһ…бҹ’бһҡбһҫбһ“бһӮбһҺбһ“бһ¶бһӮбһҳбҹ’бһӣбһ¶бһҸ бһ“бһ·бһ„бһҮбһ„бҹӢбһ”бһүбҹ’бһҲбһҡбһҠбҹ„бһҷбһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбһ·
- ** бһҖбһ¶бһҡбһ”бһ·бһ‘бһҸбһ¶бһҳбһҖбһ¶бһӣбһңбһ·бһ—бһ¶бһӮ бһ“бһ·бһ„бһҠбҹ„бһҷбһҠбҹғ **бҹ– бһӮбһ¶бҹҶбһ‘бҹ’бһҡбһҖбһ¶бһҡбһ”бһ·бһ‘бһҸбһ¶бһҳбһҖбһ¶бһӣбһңбһ·бһ—бһ¶бһӮ бһ“бһ·бһ„бһҠбҹ„бһҷбһҠбҹғ

## бһ”бҹ’бһҡбһҫбһҖбһ¶бһҡбһ”бһ„бҹ’бһ бһ¶бһү

```js
import { toast, toastErr } from "webc.site/js/Toast.js";

toast((el) => {
  el.textContent = "бһҳбһ¶бһҸбһ·бһҖбһ¶вҖӢбһҡбһ бҹҗбһҹ";
});

toastErr((el) => {
  el.textContent = "бһҳбһ¶бһҸбһ·бһҖбһ¶бһҖбҹҶбһ бһ»бһҹ";
});

toast(
  (el) => {
    el.textContent = "бһ”бһ·бһ‘бһҠбҹ„бһҷбһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбһ·бһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢбһ–бһё 99 бһңбһ·бһ“бһ¶бһ‘бһё";
  },
  99
);
```

## бһ”бҹүбһ¶бһҡбҹүбһ¶бһҳбҹүбҹӮбһҸбҹ’бһҡбһ…бҹҶбһҺбһ»бһ…бһ”бҹ’бһҡбһ‘бһ¶бһҖбҹӢ

### toast(render, timeout)

бһ”бһ„бҹ’бһҖбһҫбһҸбһ“бһ·бһ„бһ”бһ„бҹ’бһ бһ¶бһүбһҖбһ¶бһҡбһҮбһҳбҹ’бһҡбһ»бһүбҹ”

- `render`бҹ– `(el) => void` бһҖбҹҶбһ–бһ»бһ„вҖӢбһ’бҹ’бһңбһҫвҖӢбһҖбһ¶бһҡвҖӢбһ бҹ…вҖӢбһҸбҹ’бһҡбһЎбһ”бҹӢ `el` бһӮбһәвҖӢбһҮбһ¶вҖӢбһ’бһ¶бһҸбһ»вҖӢбһ”бҹ’бһҡбһўбһ”бҹӢвҖӢбһ”бһүбҹ’бһ…бһјбһӣбҹ”
- `timeout`бҹ– `Number` бһўбһҹбҹӢбһ–бҹҒбһӣбһңбһ·бһ“бһ¶бһ‘бһёбҹ” бһӣбҹҶбһ“бһ¶бҹҶбһҠбһҫбһҳ 9. бһҖбҹҶбһҺбһҸбҹӢбһ‘бҹ… 0 бһҠбһҫбһҳбҹ’бһ”бһёбһҳбһ·бһ“бһ”бһ·бһ‘бһҠбҹ„бһҷбһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбһ·бҹ”
- бһҸбһҳбҹ’бһӣбҹғбһҸбҹ’бһҡбһЎбһ”бҹӢбҹ– бһ’бһ¶бһҸбһ»бһ”бҹ’бһҡбһўбһ”бҹӢбһ”бһүбҹ’бһ…бһјбһӣбҹ” бһӮбһ¶бҹҶбһ‘бҹ’бһҡ `el.close()`бһ”бһ·бһ‘бҹ”

### toastErr(render, timeout)

бһ”бһ„бҹ’бһҖбһҫбһҸ бһ“бһ·бһ„бһ”бһ„бҹ’бһ бһ¶бһүбһҹбһ¶бһҡбһҖбҹҶбһ бһ»бһҹбҹ” бһ”бҹүбһ¶бһҡбҹүбһ¶бһҳбҹүбҹӮбһҸбҹ’бһҡвҖӢбһӮбһәвҖӢбһҠбһјбһ…вҖӢбһӮбҹ’бһ“бһ¶вҖӢбһ“бһ№бһ„ `toast` бһҠбҹ„бһҷвҖӢбһҳбһ¶бһ“ `.ERR`бһҡбһ…бһ“бһ¶бһ”бҹҗбһ‘бҹ’бһҳвҖӢбһҗбҹ’бһ“бһ¶бһҖбҹӢбҹ”

## бһҗбҹ’бһ“бһ¶бһҖбҹӢбһҡбһ…бһ“бһ¶бһ”бҹҗбһ‘бҹ’бһҳ

- `.Toast`бҹ– бһ”бҹ’бһҡбһўбһ”бҹӢбһ”бһүбҹ’бһ…бһјбһӣбһҗбҹ’бһ“бһ¶бһҖбҹӢбһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бҹ”
- `.ERR`бҹ– бһҹбҹ’бһҗбһ¶бһ“бһ—бһ¶бһ–бһҖбҹҶбһ бһ»бһҹбҹ”
- `.animated` / `.fadeInR` / `.fadeOutR`бҹ– бһҖбһ¶бһҡбһ•бҹ’бһӣбһ¶бһҹбҹӢбһ”бҹ’бһҸбһјбһҡбһҳбһ¶бһ“бһ…бһӣбһ“бһ¶бҹ”
- `.x`бҹ– бһ”бҹҠбһјбһҸбһ»бһ„бһ”бһ·бһ‘бҹ”
