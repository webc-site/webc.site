# бһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһӮбҹ’бһ“бһ¶бһ“бҹғбһ”бҹ’бһҡбһўбһ”бҹӢбһҮбҹ’бһҡбһҫбһҹбһҡбһҫбһҹбһ‘бһҳбҹ’бһӣбһ¶бһҖбҹӢбһ…бһ»бҹҮ бһ“бһ·бһ„бһ”бҹҠбһјбһҸбһ»бһ„

- бһ”бҹҶбһ”бҹӮбһҖбһ”бҹ’бһҡбһўбһ”бҹӢбһҮбҹ’бһҡбһҫбһҹбһҡбһҫбһҹ бһ“бһ·бһ„бһ”бҹҠбһјбһҸбһ»бһ„бһ•бҹ’бһҠбҹҒбһҖ
- бһҡбһҖбҹ’бһҹбһ¶бһ‘бһ»бһҖбһҹбҹ’бһҗбһ¶бһ“бһ—бһ¶бһ–бһҠбҹӮбһӣбһ”бһ¶бһ“бһҮбҹ’бһҡбһҫбһҹбһҡбһҫбһҹбһҠбҹ„бһҷбһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбһ·бһ‘бҹ…бһҖбһ“бҹ’бһӣбҹӮбһ„бһ•бҹ’бһ‘бһ»бһҖбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷ
- бһҖбһ¶бһҡбһ…бһ»бһ…бһҹбҹ’бһ‘бһ¶бһҖбҹӢбһ…бһ¶бһ”бҹӢбһ”бҹҠбһјбһҸбһ»бһ„ бһ“бһ·бһ„бһ”бһүбҹ’бһҮбһјбһ“бһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚбһ•бҹ’бһ‘бһ¶бһӣбҹӢбһҒбҹ’бһӣбһҪбһ“

## бһ”бҹ’бһҡбһҫбһҖбһ¶бһҡбһ”бһ„бҹ’бһ бһ¶бһү

```html
<c-select-btn key="my-select-key" value="option1">
  <c-select>
    <option value="option1">бһҮбһҳбҹ’бһҡбһҫбһҹбһ‘бһё 1</option>
    <option value="option2">бһҮбһҳбҹ’бһҡбһҫбһҹбһ‘бһё 2</option>
  </c-select>
  <button>бһҡбһҸбҹӢ</button>
</c-select-btn>

<script type="module">
  import "webc.site/SelectBtn.js";
  const el = document.querySelector("c-select-btn");
  el.addEventListener("submit", (e) => {
    console.log(e.value);
  });
</script>
```

## бһ‘бҹ’бһҡбһ–бҹ’бһҷбһҹбһҳбҹ’бһ”бһҸбҹ’бһҸбһ·

- `key`бҹ– бһҲбҹ’бһҳбҹ„бҹҮвҖӢбһӮбһ“бҹ’бһӣбһ№бҹҮвҖӢбһ”бҹ’бһҡбһҫвҖӢбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢвҖӢбһҖбһ¶бһҡвҖӢбһҡбһҖбҹ’бһҹбһ¶вҖӢбһ‘бһ»бһҖвҖӢбһҖбҹ’бһ“бһ»бһ„вҖӢбһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“
- `value`бҹ– бһҸбһҳбҹ’бһӣбҹғбһҠбҹӮбһӣбһ”бһ¶бһ“бһҮбҹ’бһҡбһҫбһҹбһҡбһҫбһҹбһ”бһ…бҹ’бһ…бһ»бһ”бҹ’бһ”бһ“бҹ’бһ“

## бһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚ

- `submit`бҹ– бһ”бһ¶бһ“бһ”бһүбҹ’бһҮбһјбһ“бһ“бҹ…бһ–бҹҒбһӣбһ”бҹҠбһјбһҸбһ»бһ„бһҸбҹ’бһҡбһјбһңбһ”бһ¶бһ“бһ…бһ»бһ… `e.value` бһ•бҹ’бһ‘бһ»бһҖбһҸбһҳбҹ’бһӣбҹғбһҠбҹӮбһӣбһ”бһ¶бһ“бһҮбҹ’бһҡбһҫбһҹбһҡбһҫбһҹбһ”бһ…бҹ’бһ…бһ»бһ”бҹ’бһ”бһ“бҹ’бһ“
