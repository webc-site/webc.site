# ಮಾರ್ಕ್‌ಡೌನ್ ಪಠ್ಯದ ಸ್ಟ್ರೀಮಿಂಗ್ ರೆಂಡರಿಂಗ್

- ಹೆಚ್ಚುತ್ತಿರುವಂತೆ ಪಾರ್ಸ್ ಮಾಡಿ ಮತ್ತು ಮಾರ್ಕ್‌ಡೌನ್ ಅನ್ನು ನಿರೂಪಿಸಿ
- ವಿಷಯವು ಪರದೆಯನ್ನು ಮೀರಿದಾಗ ಸ್ವಯಂಚಾಲಿತವಾಗಿ ಸ್ಕ್ರಾಲ್ ಮಾಡಲು ಸ್ಕ್ರೋಲಿಂಗ್ ಮೂಲ ಅಂಶದೊಂದಿಗೆ ಸಹಕರಿಸಿ
- ಕ್ಲಿಕ್ ಮಾಡುವಾಗ ಅಥವಾ ಮೇಲಕ್ಕೆ ಸ್ಕ್ರಾಲ್ ಮಾಡುವಾಗ ಸ್ವಯಂ ಸ್ಕ್ರಾಲ್ ಅನ್ನು ವಿರಾಮಗೊಳಿಸಿ, ಕೆಳಗೆ ಸ್ಕ್ರಾಲ್ ಮಾಡುವಾಗ ಪುನರಾರಂಭಿಸಿ
- ಟೈಪಿಂಗ್ ಸ್ಥಿತಿ ಸೂಚಕಗಳನ್ನು ಸ್ವಯಂಚಾಲಿತವಾಗಿ ಪ್ರದರ್ಶಿಸಿ

## ಡೆಮೊ ಬಳಸಿ

```html
<c-vs>
  <c-md-stream id="md-view"></c-md-stream>
</c-vs>

<script type="module">
  import "webc.site/MdStream.js";

  const el = document.getElementById("md-view"),
    sleep = (ms) => new Promise((resolve) => setTimeout(resolve, ms));

  // ಹಿಂತಿರುಗಿದ ಅಸಮಕಾಲಿಕ ಜನರೇಟರ್ ಕಾರ್ಯದ ಮೂಲಕ ಸ್ಟ್ರೀಮಿಂಗ್ ಡೇಟಾ ರೆಂಡರಿಂಗ್
  el.gen = async function* () {
    ಇಳುವರಿ "# ಶೀರ್ಷಿಕೆ\n";
    for (let i = 1; i <= 100; ++i) {
      await sleep(400);
      yield "- ಅಂಶ " + i + "\n";
    }
  };
</script>
```

## ಆಸ್ತಿ

| ಗುಣಲಕ್ಷಣದ ಹೆಸರು | ರೀತಿಯ | ವಿವರಿಸಲು |
| :--- | :--- | :--- |
| `gen` | `Function` | ಅಸಮಕಾಲಿಕ ಪುನರಾವರ್ತಕ/ಜನರೇಟರ್ ಅನ್ನು ಹಿಂತಿರುಗಿಸುವ ಕಾರ್ಯ, ವಿಷಯವನ್ನು ತೆರವುಗೊಳಿಸುತ್ತದೆ ಮತ್ತು ಸೆಟ್ಟಿಂಗ್ ನಂತರ ಮರು-ರೆಂಡರ್ ಮಾಡುತ್ತದೆ |
